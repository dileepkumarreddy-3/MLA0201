import pandas as pd
from sklearn.linear_model import LinearRegression

data = pd.read_csv("sales.csv")

X = data[['Month']]
y = data['Sales']

model = LinearRegression()
model.fit(X, y)

future_month = [[13]]
future_sales = model.predict(future_month)
print("Predicted Future Sales:", future_sales[0])
output
Predicted Future Sales: 5500
